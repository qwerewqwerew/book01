const btn = $('ul>li'), panel = $('h3');
hidden(panel);
visible(panel.eq(0));
btn.each(function (i, o) {
  btn.eq(i).click(function () {
    hidden(panel);
    visible(panel.eq(i));
  })
})
function visible(obj) {
  obj.show();
}
function hidden(obj) {
  obj.hide();
}
